-_-_-_-_-_-_-_-_-_-_-_- Switch -_-_-_-_-_-_-_-_-_-_-_-

Syntax:-
	
	switch(variableName){
		case value:
			// if matched with value further procedure is here
			break;
		case other_value:
			// if matched with other_value further procedure is here
			break;
		default:
			// if nothing matches further procedure is here
	}
